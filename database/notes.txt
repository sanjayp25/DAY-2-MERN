  update operators:
    1.$set
    2.$unset
    3.$pop
    4.$push
    5.$inc

    Syntax:
     db.collectionName.updateOne({filter condition},{updated query})
     db.collectionName.updateOne({filter condition},{$set:{updated query}}) add new data or update
     db.collectionName.updateOne({filter condition},{$unset:{updated query}}) delete field
     db.collectionName.updateOne({filter condition},{$inc:{updated query}}) increment numbers with existing numbers
     db.collectionName.updateOne({filter condition},{$inc:{updated query}}) include (-)symbol with numbers add with existing numbers
     db.collectionName.updateOne({filter condition},{$push:{updated query}}) add value in existinf field array
